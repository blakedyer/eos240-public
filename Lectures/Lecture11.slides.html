def batch_S(F, D, Co): #batch melting equation for solid
    return (D*Co) / (F + D * (1 - F))

D_n = np.array([1,2,3,4]) #a list of partition coefficients
F = 0.8 #melt fraction
N_0 = batch_S(F, D_n, 1) #C_l for each F
